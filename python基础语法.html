装饰器

功能：给原函数添加新的功能
特点，原则：

  1. 不改变原函数
  2. 不改变原函数的调用方式
  3. 同时满足1，2条件
import time
def fn():
    time.sleep(1)
    print("hello world")
def fn1():
    time.sleep(1)
    print("hello")
def fn2(fn):
    def newFn():
        start = time.time()
        fn()
        end = time.time()
        print(f"总时间为:{end-start}")
    return newFn
fn=fn2(fn)
fn1=fn2(fn1)
fn()
fn1()

万能传参

staus = 0 #没有登陆 1 登录

def login():
    global staus
    print("登录页面")
    username = input("请输入用户名") #str
    password = input("请输入密码")
    if username == "123" and password=="123456":
        staus = 1
        print("登录成功")
    else:
        print("登录失败")


def authLogin(fn):
    def newFn(*args,**kwargs):#万能传参
        if staus == 1:
            fn(*args,**kwargs)
        else:
            login()
    return newFn

@authLogin
def fn1(id):
    print("成绩查看页面%s"%id)
@authLogin
def fn2(id):
    print("课程查看页面%s"%id)

login()
fn1(10)
fn2(10)
